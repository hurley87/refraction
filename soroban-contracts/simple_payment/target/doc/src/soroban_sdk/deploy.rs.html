1//! Deploy contains types for deploying contracts.
2//!
3//! Contracts are assigned an ID that is derived from a set of arguments. A
4//! contract may choose which set of arguments to use to deploy with:
5//!
6//! - [Deployer::with_current_contract] â€“ A contract deployed by the currently
7//! executing contract will have an ID derived from the currently executing
8//! contract's ID.
9//!
10//! The deployer can be created using [Env::deployer].
11//!
12//! ### Examples
13//!
14//! ```
15//! # use soroban_sdk::{contract, contractimpl, BytesN, Env, Symbol};
16//! #
17//! # #[contract]
18//! # pub struct Contract;
19//! #
20//! # #[contractimpl]
21//! # impl Contract {
22//! #     pub fn f(env: Env, wasm_hash: BytesN<32>) {
23//! #         let salt = [0u8; 32];
24//! #         let deployer = env.deployer().with_current_contract(salt);
25//! #         // Deployed contract address is deterministic and can be accessed
26//! #         // before deploying the contract.
27//! #         let _ = deployer.deployed_address();
28//! #         let contract_address = deployer.deploy(wasm_hash);
29//! #     }
30//! # }
31//! #
32//! # #[cfg(feature = "testutils")]
33//! # fn main() {
34//! #     let env = Env::default();
35//! #     let contract_address = env.register_contract(None, Contract);
36//! #     // Install the contract code before deploying its instance.
37//! #     let mock_wasm = [0u8; 0];
38//! #     let wasm_hash = env.deployer().upload_contract_wasm(mock_wasm.as_slice());
39//! #     ContractClient::new(&env, &contract_address).f(&wasm_hash);
40//! # }
41//! # #[cfg(not(feature = "testutils"))]
42//! # fn main() { }
43//! ```
44

46    env::internal::Env as _, unwrap::UnwrapInfallible, Address, Bytes, BytesN, Env, IntoVal,

49/// Deployer provides access to deploying contracts.
50pub struct Deployer {
51    env: Env,
52}
53
54impl Deployer {
55    pub(crate) fn new(env: &Env) -> Deployer {
56        Deployer { env: env.clone() }
57    }
58
59    pub fn env(&self) -> &Env {
60        &self.env
61    }
62
63    /// Get a deployer that deploys contract that derive the contract IDs
64    /// from the current contract and provided salt.
65    pub fn with_current_contract(
66        &self,
67        salt: impl IntoVal<Env, BytesN<32>>,
68    ) -> DeployerWithAddress {
69        DeployerWithAddress {
70            env: self.env.clone(),
71            address: self.env.current_contract_address(),
72            salt: salt.into_val(&self.env),
73        }
74    }
75
76    /// Get a deployer that deploys contracts that derive the contract ID
77    /// from the provided address and salt.
78    ///
79    /// The deployer address must authorize all the deployments.
80    pub fn with_address(
81        &self,
82        address: Address,
83        salt: impl IntoVal<Env, BytesN<32>>,
84    ) -> DeployerWithAddress {
85        DeployerWithAddress {
86            env: self.env.clone(),
87            address,
88            salt: salt.into_val(&self.env),
89        }
90    }
91
92    /// Get a deployer that deploys an instance of Stellar Asset Contract
93    /// corresponding to the provided serialized asset.
94    ///
95    /// `serialized_asset` is the Stellar `Asset` XDR serialized to bytes. Refer
96    /// to `[soroban_sdk::xdr::Asset]`
97    pub fn with_stellar_asset(
98        &self,
99        serialized_asset: impl IntoVal<Env, Bytes>,
100    ) -> DeployerWithAsset {
101        DeployerWithAsset {
102            env: self.env.clone(),
103            serialized_asset: serialized_asset.into_val(&self.env),
104        }
105    }
106
107    /// Upload the contract Wasm code to the network.
108    ///
109    /// Returns the hash of the uploaded Wasm that can be then used for
110    /// the contract deployment.
111    /// ### Examples
112    /// ```
113    /// use soroban_sdk::{BytesN, Env};
114    ///
115    /// const WASM: &[u8] = include_bytes!("../doctest_fixtures/contract.wasm");
116    ///
117    /// #[test]
118    /// fn test() {
119    /// # }
120    /// # fn main() {
121    ///     let env = Env::default();
122    ///     env.deployer().upload_contract_wasm(WASM);
123    /// }
124    /// ```
125    pub fn upload_contract_wasm(&self, contract_wasm: impl IntoVal<Env, Bytes>) -> BytesN<32> {
126        self.env
127            .upload_wasm(contract_wasm.into_val(&self.env).to_object())
128            .unwrap_infallible()
129            .into_val(&self.env)
130    }
131
132    /// Replaces the executable of the current contract with the provided Wasm.
133    ///
134    /// The Wasm blob identified by the `wasm_hash` has to be already present
135    /// in the ledger (uploaded via `[Deployer::upload_contract_wasm]`).
136    ///
137    /// The function won't do anything immediately. The contract executable
138    /// will only be updated after the invocation has successfully finished.
139    pub fn update_current_contract_wasm(&self, wasm_hash: impl IntoVal<Env, BytesN<32>>) {
140        self.env
141            .update_current_contract_wasm(wasm_hash.into_val(&self.env).to_object())
142            .unwrap_infallible();
143    }
144
145    /// Extend the TTL of the contract instance and code.
146    ///
147    /// Extends the TTL of the instance and code only if the TTL for the provided contract is below `threshold` ledgers.
148    /// The TTL will then become `extend_to`. Note that the `threshold` check and TTL extensions are done for both the
149    /// contract code and contract instance, so it's possible that one is bumped but not the other depending on what the
150    /// current TTL's are.
151    ///
152    /// The TTL is the number of ledgers between the current ledger and the final ledger the data can still be accessed.
153    pub fn extend_ttl(&self, contract_address: Address, threshold: u32, extend_to: u32) {
154        self.env
155            .extend_contract_instance_and_code_ttl(
156                contract_address.to_object(),
157                threshold.into(),
158                extend_to.into(),
159            )
160            .unwrap_infallible();
161    }
162
163    /// Extend the TTL of the contract instance.
164    ///
165    /// Same as [`extend_ttl`](Self::extend_ttl) but only for contract instance.
166    pub fn extend_ttl_for_contract_instance(
167        &self,
168        contract_address: Address,
169        threshold: u32,
170        extend_to: u32,
171    ) {
172        self.env
173            .extend_contract_instance_ttl(
174                contract_address.to_object(),
175                threshold.into(),
176                extend_to.into(),
177            )
178            .unwrap_infallible();
179    }
180
181    /// Extend the TTL of the contract code.
182    ///
183    /// Same as [`extend_ttl`](Self::extend_ttl) but only for contract code.
184    pub fn extend_ttl_for_code(&self, contract_address: Address, threshold: u32, extend_to: u32) {
185        self.env
186            .extend_contract_code_ttl(
187                contract_address.to_object(),
188                threshold.into(),
189                extend_to.into(),
190            )
191            .unwrap_infallible();
192    }
193}

195/// A deployer that deploys a contract that has its ID derived from the provided
196/// address and salt.
197pub struct DeployerWithAddress {
198    env: Env,
199    address: Address,
200    salt: BytesN<32>,
201}
202
203impl DeployerWithAddress {
204    /// Return the address of the contract defined by the deployer.
205    ///
206    /// This function can be called at anytime, before or after the contract is
207    /// deployed, because contract addresses are deterministic.
208    pub fn deployed_address(&self) -> Address {
209        self.env
210            .get_contract_id(self.address.to_object(), self.salt.to_object())
211            .unwrap_infallible()
212            .into_val(&self.env)
213    }
214
215    /// Deploy a contract that uses Wasm executable with provided hash.
216    ///
217    /// The address of the deployed contract is defined by the deployer address
218    /// and provided salt.
219    ///
220    /// Returns the deployed contract's address.
221    pub fn deploy(&self, wasm_hash: impl IntoVal<Env, BytesN<32>>) -> Address {
222        let env = &self.env;
223        let address_obj = env
224            .create_contract(
225                self.address.to_object(),
226                wasm_hash.into_val(env).to_object(),
227                self.salt.to_object(),
228            )
229            .unwrap_infallible();
230        unsafe { Address::unchecked_new(env.clone(), address_obj) }
231    }
232}
233
234pub struct DeployerWithAsset {
235    env: Env,
236    serialized_asset: Bytes,
237}
238
239impl DeployerWithAsset {
240    /// Return the address of the contract defined by the deployer.
241    ///
242    /// This function can be called at anytime, before or after the contract is
243    /// deployed, because contract addresses are deterministic.
244    pub fn deployed_address(&self) -> Address {
245        self.env
246            .get_asset_contract_id(self.serialized_asset.to_object())
247            .unwrap_infallible()
248            .into_val(&self.env)
249    }
250
251    pub fn deploy(&self) -> Address {
252        self.env
253            .create_asset_contract(self.serialized_asset.to_object())
254            .unwrap_infallible()
255            .into_val(&self.env)
256    }
257}

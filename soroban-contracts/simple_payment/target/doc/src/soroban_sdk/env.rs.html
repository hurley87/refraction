27    // When we have `feature="testutils"` (or are in cfg(test)) we enable feature
28    // `soroban-env-{common,host}/testutils` which in turn adds the helper method
29    // `Env::escalate_error_to_panic` to the Env trait.
30    //
31    // When this is available we want to use it, because it works in concert
32    // with a _different_ part of the host that's also `testutils`-gated: the
33    // mechanism for emulating the WASM VM error-handling semantics with native
34    // contracts. In particular when a WASM contract calls a host function that
35    // fails with some error E, the host traps the VM (not returning to it at
36    // all) and propagates E to the caller of the contract. This is simulated in
37    // the native case by returning a (nontrivial) error E to us here, which we
38    // then "reject" back to the host, which stores E in a temporary cell inside
39    // any `TestContract` frame in progress and then _panics_, unwinding back to
40    // a panic-catcher it installed when invoking the `TestContract` frame, and
41    // then extracting E from the frame and returning it to its caller. This
42    // simulates the "crash, but catching the error" behaviour of the WASM case.
43    // This only works if we panic via `escalate_error_to_panic`.
44    //
45    // (The reason we don't just panic_any() here and let the panic-catcher do a
46    // type-based catch is that there might _be_ no panic-catcher around us, and
47    // we want to print out a nice error message in that case too, which
48    // panic_any() does not do us the favor of producing. This is all very
49    // subtle. See also soroban_env_host::Host::escalate_error_to_panic.)
50    #[cfg(any(test, feature = "testutils"))]
51    pub(crate) fn reject_err<T>(env: &Host, r: Result<T, HostError>) -> Result<T, Infallible> {
52        r.map_err(|e| env.escalate_error_to_panic(e))
53    }
54
55    // When we're _not_ in a cfg enabling `soroban-env-{common,host}/testutils`,
56    // there is no `Env::escalate_error_to_panic` to call, so we just panic
57    // here. But this is ok because in that case there is also no multi-contract
58    // calling machinery set up, nor probably any panic-catcher installed that
59    // we need to hide error values for the benefit of. Any panic in this case
60    // is probably going to unwind completely anyways. No special case needed.
61    #[cfg(not(any(test, feature = "testutils")))]
62    pub(crate) fn reject_err<T>(_env: &Host, r: Result<T, HostError>) -> Result<T, Infallible> {
63        r.map_err(|e| panic!("{:?}", e))
64    }

358    ///
359    /// # Panics
360    ///
361    /// Will panic if the `contract_id` does not match a registered contract,

387            .map_err(|_| ConversionError)
388            .unwrap()
389    }

392    /// returns an error if the invocation fails for any reason.

394        &self,
395        contract_address: &Address,
396        func: &crate::Symbol,
397        args: Vec<Val>,
398    ) -> Result<Result<T, T::Error>, Result<E, InvokeError>>

412            Ok(err) => Err(E::try_from(err).map_err(Into::into)),
413            Err(ConversionError) => Ok(T::try_from_val(self, &rv)),

579    /// Register a contract with the [Env] for testing.
580    ///
581    /// Passing a contract ID for the first arguments registers the contract
582    /// with that contract ID. Providing `None` causes the Env to generate a new
583    /// contract ID that is assigned to the contract.
584    ///
585    /// Registering a contract that is already registered replaces it.
586    ///
587    /// Returns the contract ID of the registered contract.
588    ///
589    /// ### Examples
590    /// ```
591    /// use soroban_sdk::{contract, contractimpl, BytesN, Env, Symbol};
592    ///
593    /// #[contract]
594    /// pub struct HelloContract;
595    ///
596    /// #[contractimpl]
597    /// impl HelloContract {
598    ///     pub fn hello(env: Env, recipient: Symbol) -> Symbol {
599    ///         todo!()
600    ///     }
601    /// }
602    ///
603    /// #[test]
604    /// fn test() {
605    /// # }
606    /// # fn main() {
607    ///     let env = Env::default();
608    ///     let contract_id = env.register_contract(None, HelloContract);
609    /// }
610    /// ```
611    pub fn register_contract<'a, T: ContractFunctionSet + 'static>(
612        &self,
613        contract_id: impl Into<Option<&'a Address>>,
614        contract: T,
615    ) -> Address {
616        struct InternalContractFunctionSet<T: ContractFunctionSet>(pub(crate) T);
617        impl<T: ContractFunctionSet> internal::ContractFunctionSet for InternalContractFunctionSet<T> {
618            fn call(
619                &self,
620                func: &Symbol,
621                env_impl: &internal::EnvImpl,
622                args: &[Val],
623            ) -> Option<Val> {
624                let env = Env {
625                    env_impl: env_impl.clone(),
626                    test_state: Default::default(),
627                };
628                self.0.call(
629                    crate::Symbol::try_from_val(&env, func)
630                        .unwrap_infallible()
631                        .to_string()
632                        .as_str(),
633                    env,
634                    args,
635                )
636            }
637        }
638
639        let contract_id = if let Some(contract_id) = contract_id.into() {
640            contract_id.clone()
641        } else {
642            Address::generate(self)
643        };
644        self.env_impl
645            .register_test_contract(
646                contract_id.to_object(),
647                Rc::new(InternalContractFunctionSet(contract)),
648            )
649            .unwrap();
650        contract_id
651    }

1109    /// Invokes the special `__check_auth` function of contracts that implement
1110    /// the custom account interface.
1111    ///
1112    /// `__check_auth` can't be called outside of the host-managed `require_auth`
1113    /// calls. This test utility allows testing custom account contracts without
1114    /// the need to setup complex contract call trees and enabling the enforcing
1115    /// auth on the host side.
1116    ///
1117    /// This function requires to provide the template argument for error. Use
1118    /// `soroban_sdk::Error` if `__check_auth` doesn't return a special
1119    /// contract error and use the error with `contracterror` attribute
1120    /// otherwise.
1121    ///
1122    /// ### Examples
1123    /// ```
1124    /// use soroban_sdk::{contract, contracterror, contractimpl, testutils::{Address as _, BytesN as _}, vec, auth::Context, BytesN, Env, Vec, Val};
1125    ///
1126    /// #[contracterror]
1127    /// #[derive(Copy, Clone, Debug, Eq, PartialEq, PartialOrd, Ord)]
1128    /// #[repr(u32)]
1129    /// pub enum NoopAccountError {
1130    ///     SomeError = 1,
1131    /// }
1132    /// #[contract]
1133    /// struct NoopAccountContract;
1134    /// #[contractimpl]
1135    /// impl NoopAccountContract {
1136    ///
1137    ///     #[allow(non_snake_case)]
1138    ///     pub fn __check_auth(
1139    ///         _env: Env,
1140    ///         _signature_payload: BytesN<32>,
1141    ///         signature: Val,
1142    ///         _auth_context: Vec<Context>,
1143    ///     ) -> Result<(), NoopAccountError> {
1144    ///         if signature.is_void() {
1145    ///             Err(NoopAccountError::SomeError)
1146    ///         } else {
1147    ///             Ok(())
1148    ///         }
1149    ///     }
1150    /// }
1151    /// #[test]
1152    /// fn test() {
1153    /// # }
1154    /// # fn main() {
1155    ///     let e: Env = Default::default();
1156    ///     let account_contract = NoopAccountContractClient::new(&e, &e.register_contract(None, NoopAccountContract));
1157    ///     // Non-succesful call of `__check_auth` with a `contracterror` error.
1158    ///     assert_eq!(
1159    ///         e.try_invoke_contract_check_auth::<NoopAccountError>(
1160    ///             &account_contract.address,
1161    ///             &BytesN::from_array(&e, &[0; 32]),
1162    ///             ().into(),
1163    ///             &vec![&e],
1164    ///         ),
1165    ///         // The inner `Result` is for conversion error and will be Ok
1166    ///         // as long as a valid error type used.
1167    ///         Err(Ok(NoopAccountError::SomeError))
1168    ///     );
1169    ///     // Successful call of `__check_auth` with a `soroban_sdk::InvokeError`
1170    ///     // error - this should be compatible with any error type.
1171    ///     assert_eq!(
1172    ///         e.try_invoke_contract_check_auth::<soroban_sdk::InvokeError>(
1173    ///             &account_contract.address,
1174    ///             &BytesN::from_array(&e, &[0; 32]),
1175    ///             0_i32.into(),
1176    ///             &vec![&e],
1177    ///         ),
1178    ///         Ok(())
1179    ///     );
1180    /// }
1181    /// ```
1182    pub fn try_invoke_contract_check_auth<E>(
1183        &self,
1184        contract: &Address,
1185        signature_payload: &BytesN<32>,
1186        signature: Val,
1187        auth_context: &Vec<auth::Context>,
1188    ) -> Result<(), Result<E, InvokeError>>
1189    where
1190        E: TryFrom<Error>,
1191        E::Error: Into<InvokeError>,
1192    {
1193        let args = Vec::from_array(
1194            self,
1195            [signature_payload.to_val(), signature, auth_context.to_val()],
1196        );
1197        let res = self
1198            .host()
1199            .call_account_contract_check_auth(contract.to_object(), args.to_object());
1200        match res {
1201            Ok(rv) => Ok(rv.into_val(self)),
1202            Err(e) => Err(e.error.try_into().map_err(Into::into)),
1203        }
1204    }

1403#[cfg(any(test, feature = "testutils"))]
1404impl Env {
1405    /// Create a snapshot file for the currently executing test.
1406    ///
1407    /// Writes the file to the `test_snapshots/{test-name}.N.json` path where
1408    /// `N` is incremented for each unique `Env` in the test.
1409    ///
1410    /// Use to record the observable behavior of a test, and changes to that
1411    /// behavior over time. Commit the test snapshot file to version control and
1412    /// watch for changes in it on contract change, SDK upgrade, protocol
1413    /// upgrade, and other important events.
1414    ///
1415    /// No file will be created if the environment has no meaningful data such
1416    /// as stored entries or events.
1417    ///
1418    /// ### Panics
1419    ///
1420    /// If there is any error writing the file.
1421    pub(crate) fn to_test_snapshot_file(&self) {
1422        let snapshot = self.to_snapshot();
1423
1424        // Don't write a snapshot that has no data in it.
1425        if snapshot.ledger.entries().into_iter().count() == 0
1426            && snapshot.events.0.is_empty()
1427            && snapshot.auth.0.is_empty()
1428        {
1429            return;
1430        }
1431
1432        // Determine path to write test snapshots to.
1433        let thread = std::thread::current();
1434        let Some(test_name) = thread.name() else {
1435            // The stock unit test runner sets a thread name.
1436            // If there is no thread name, assume this is not running as
1437            // part of a unit test, and do nothing.
1438            return;
1439        };
1440        if test_name == "main" {
1441            // When doc tests are running they're all run with the thread name
1442            // main. There's no way to detect which doc test is being run and
1443            // there's little value in writing and overwriting a single file for
1444            // all doc tests.
1445            return;
1446        }
1447        let file_number = LAST_TEST_SNAPSHOT.with_borrow_mut(|l| {
1448            if test_name == l.name {
1449                *l = LastTestSnapshot::default();
1450                l.name = test_name.to_owned();
1451            }
1452            l.number += 1;
1453            l.number
1454        });
1455        // Break up the test name into directories, using :: as the separator.
1456        // The :: module separator cannot be written into the filename because
1457        // some operating systems (e.g. Windows) do not allow the : character in
1458        // filenames.
1459        let test_name_path = test_name
1460            .split("::")
1461            .map(|p| std::path::Path::new(p).to_path_buf())
1462            .reduce(|p0, p1| p0.join(p1))
1463            .expect("test name to not be empty");
1464        let dir = std::path::Path::new("test_snapshots");
1465        let p = dir
1466            .join(&test_name_path)
1467            .with_extension(format!("{file_number}.json"));
1468
1469        // Write test snapshots to file.
1470        eprintln!("Writing test snapshot file for test {test_name:?} to {p:?}.");
1471        snapshot.write_file(p).unwrap();
1472    }
1473}

1654// This is a helper macro used only by impl_env_for_sdk below. It consumes a
1655// token-tree of the form:
1656//
1657//  {fn $fn_id:ident $args:tt -> $ret:ty}
1658//
1659// and produces the the corresponding method definition to be used in the
1660// SDK's Env implementation of the Env (calling through to the corresponding
1661// guest or host implementation).
1662macro_rules! sdk_function_helper {
1663    {$mod_id:ident, fn $fn_id:ident($($arg:ident:$type:ty),*) -> $ret:ty}
1664    =>
1665    {
1666        fn $fn_id(&self, $($arg:$type),*) -> Result<$ret, Self::Error> {
1667            internal::reject_err(&self.env_impl, self.env_impl.$fn_id($($arg),*))
1668        }
1669    };
1670}
1671
1672// This is a callback macro that pattern-matches the token-tree passed by the
1673// x-macro (call_macro_with_all_host_functions) and produces a suite of
1674// forwarding-method definitions, which it places in the body of the declaration
1675// of the implementation of Env for the SDK's Env.
1676macro_rules! impl_env_for_sdk {
1677    {
1678        $(
1679            // This outer pattern matches a single 'mod' block of the token-tree
1680            // passed from the x-macro to this macro. It is embedded in a `$()*`
1681            // pattern-repetition matcher so that it will match all provided
1682            // 'mod' blocks provided.
1683            $(#[$mod_attr:meta])*
1684            mod $mod_id:ident $mod_str:literal
1685            {
1686                $(
1687                    // This inner pattern matches a single function description
1688                    // inside a 'mod' block in the token-tree passed from the
1689                    // x-macro to this macro. It is embedded in a `$()*`
1690                    // pattern-repetition matcher so that it will match all such
1691                    // descriptions.
1692                    $(#[$fn_attr:meta])*
1693                    { $fn_str:literal, $($min_proto:literal)?, $($max_proto:literal)?, fn $fn_id:ident $args:tt -> $ret:ty }
1694                )*
1695            }
1696        )*
1697    }
1698
1699    =>  // The part of the macro above this line is a matcher; below is its expansion.
1700
1701    {
1702        // This macro expands to a single item: the implementation of Env for
1703        // the SDK's Env struct used by client contract code running in a WASM VM.
1704        #[doc(hidden)]
1705        impl internal::Env for Env
1706        {
1707            $(
1708                $(
1709                    // This invokes the guest_function_helper! macro above
1710                    // passing only the relevant parts of the declaration
1711                    // matched by the inner pattern above. It is embedded in two
1712                    // nested `$()*` pattern-repetition expanders that
1713                    // correspond to the pattern-repetition matchers in the
1714                    // match section, but we ignore the structure of the 'mod'
1715                    // block repetition-level from the outer pattern in the
1716                    // expansion, flattening all functions from all 'mod' blocks
1717                    // into the implementation of Env for Guest.
1718                    sdk_function_helper!{$mod_id, fn $fn_id $args -> $ret}
1719                )*
1720            )*
1721        }
1722    };
1723}

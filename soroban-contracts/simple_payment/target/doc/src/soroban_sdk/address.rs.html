14/// Address is a universal opaque identifier to use in contracts.
15///
16/// Address can be used as an input argument (for example, to identify the
17/// payment recipient), as a data key (for example, to store the balance), as
18/// the authentication & authorization source (for example, to authorize the
19/// token transfer) etc.
20///
21/// See `require_auth` documentation for more details on using Address for
22/// authorization.
23///
24/// Internally, Address may represent a Stellar account or a contract. Contract
25/// address may be used to identify the account contracts - special contracts
26/// that allow customizing authentication logic and adding custom authorization
27/// rules.
28///
29/// In tests Addresses should be generated via `Address::generate()`.
30#[derive(Clone)]
31pub struct Address {
32    env: Env,
33    obj: AddressObject,
34}

36impl Debug for Address {
37    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
38        #[cfg(target_family = "wasm")]
39        write!(f, "Address(..)")?;
40        #[cfg(not(target_family = "wasm"))]
41        {
42            use crate::env::internal::xdr;
43            use stellar_strkey::{ed25519, Contract, Strkey};
44            let sc_val = ScVal::try_from(self).map_err(|_| core::fmt::Error)?;
45            if let ScVal::Address(addr) = sc_val {
46                match addr {
47                    xdr::ScAddress::Account(account_id) => {
48                        let xdr::AccountId(xdr::PublicKey::PublicKeyTypeEd25519(xdr::Uint256(
49                            ed25519,
50                        ))) = account_id;
51                        let strkey = Strkey::PublicKeyEd25519(ed25519::PublicKey(ed25519));
52                        write!(f, "AccountId({})", strkey.to_string())?;
53                    }
54                    xdr::ScAddress::Contract(contract_id) => {
55                        let strkey = Strkey::Contract(Contract(contract_id.0));
56                        write!(f, "Contract({})", strkey.to_string())?;
57                    }
58                }
59            } else {
60                return Err(core::fmt::Error);
61            }
62        }
63        Ok(())
64    }
65}

81impl Ord for Address {
82    fn cmp(&self, other: &Self) -> Ordering {
83        self.env.check_same_env(&other.env).unwrap_infallible();
84        let v = self
85            .env
86            .obj_cmp(self.obj.to_val(), other.obj.to_val())

187impl Address {
188    /// Ensures that this Address has authorized invocation of the current
189    /// contract with the provided arguments.
190    ///
191    /// During the on-chain execution the Soroban host will perform the needed
192    /// authentication (verify the signatures) and ensure the replay prevention.
193    /// The contracts don't need to perform this tasks.
194    ///
195    /// The arguments don't have to match the arguments of the contract
196    /// invocation. However, it's considered the best practice to have a
197    /// well-defined, deterministic and ledger-state-independent mapping between
198    /// the contract invocation arguments and `require_auth` arguments. This
199    /// will allow the contract callers to easily build the required signature
200    /// payloads and prevent potential authorization failures.
201    ///
202    /// ### Panics
203    ///
204    /// If the invocation is not authorized.
205    pub fn require_auth_for_args(&self, args: Vec<Val>) {
206        self.env.require_auth_for_args(self, args);
207    }
208
209    /// Ensures that this Address has authorized invocation of the current
210    /// contract with all the invocation arguments
211    ///
212    /// This works exactly in the same fashion as `require_auth_for_args`, but
213    /// arguments are automatically inferred from the current contract
214    /// invocation.
215    ///
216    /// This is useful when there is only a single Address that needs to
217    /// authorize the contract invocation and there are no dynamic arguments
218    /// that don't need authorization.
219    ///
220    /// ### Panics
221    ///
222    /// If the invocation is not authorized.
223    pub fn require_auth(&self) {
224        self.env.require_auth(self);
225    }
226
227    /// Creates an `Address` corresponding to the provided Stellar strkey.
228    ///
229    /// The only supported strkey types are account keys (`G...`) and contract keys (`C...`). Any
230    /// other valid or invalid strkey will cause this to panic.
231    ///
232    /// Prefer using the `Address` directly as input or output argument. Only
233    /// use this in special cases when addresses need to be shared between
234    /// different environments (e.g. different chains).
235    pub fn from_string(strkey: &String) -> Self {
236        let env = strkey.env();
237        unsafe {
238            Self::unchecked_new(
239                env.clone(),
240                env.strkey_to_address(strkey.to_object().to_val())
241                    .unwrap_infallible(),
242            )
243        }
244    }
245
246    /// Creates an `Address` corresponding to the provided Stellar strkey bytes.
247    ///
248    /// This behaves exactly in the same fashion as `from_strkey`, i.e. the bytes should contain
249    /// exactly the same contents as `String` would (i.e. base-32 ASCII string).
250    ///
251    /// The only supported strkey types are account keys (`G...`) and contract keys (`C...`). Any
252    /// other valid or invalid strkey will cause this to panic.
253    ///
254    /// Prefer using the `Address` directly as input or output argument. Only
255    /// use this in special cases when addresses need to be shared between
256    /// different environments (e.g. different chains).
257    pub fn from_string_bytes(strkey: &Bytes) -> Self {
258        let env = strkey.env();
259        unsafe {
260            Self::unchecked_new(
261                env.clone(),
262                env.strkey_to_address(strkey.to_object().to_val())
263                    .unwrap_infallible(),
264            )
265        }
266    }
267
268    pub fn to_string(&self) -> String {
269        String::try_from_val(
270            &self.env,
271            &self.env.address_to_strkey(self.obj).unwrap_infallible(),
272        )
273        .unwrap_optimized()
274    }

278        Self { env, obj }
279    }

319#[cfg(not(target_family = "wasm"))]
320impl Address {
321    pub(crate) fn contract_id(&self) -> Hash {
322        let sc_address: ScAddress = self.try_into().unwrap();
323        if let ScAddress::Contract(c) = sc_address {
324            c
325        } else {
326            panic!("address is not a contract {:?}", self);
327        }
328    }
329
330    pub(crate) fn from_contract_id(env: &Env, contract_id: [u8; 32]) -> Self {
331        Self::try_from_val(env, &ScAddress::Contract(Hash(contract_id))).unwrap()
332    }
